= return


// OVERVIEW SECTION STARTS
[#overview]
--

[float]
=== 설명
함수를 마치고 원한다면 함수를 부른 함수로 값을 반환한다.
[%hardbreaks]


[float]
=== 문법
[source,arduino]
----
return;

return value; // 두가지 모두 타당
----


[float]
=== 매개변수
`value': 변수 또는 상수

--

[float]
=== 예제 코드
센서 입력을 임계치와 비교하는 함수

[source,arduino]
----
 int checkSensor(){
    if (analogRead(0) > 400) {
        return 1;
    }
    else{
        return 0;
    }
}
----
return 키워드는 버그 있을 수 있는 코드의 큰 섹션을 "주석 처리" 않고도 섹션을 테스트 하는데 편하다
[source,arduino]
----
void loop(){

// 빛나는 코드 아이디어 테스트는 여기에

return;

// 나머지 역기능 스케치는 여기에
// 이 코드는 실행 안 됨
}
----
[%hardbreaks]
